```python exec
import reflex as rx
from reflex.components.radix.themes.components import *
from reflex.components.radix.themes.layout import *
from reflex.components.radix.themes.typography import *
```


```python demo

popover_root(
    popover_trigger(
        button("Comment", variant="soft"),
    ),
    popover_content(
        flex(
            avatar(
                "2",
                fallback="RX",
                radius="full"
            ),
            box(
                textarea(placeholder="Write a comment…", style={"height": 80}),
                flex(
                    flex(
                        text(
                            checkbox(),
                            text("Send to group"),
                            as_="label",
                            size="2",
                        ),
                        align="center",
                        gap="2",
                        as_child=True,
                    ),
                    popover_close(
                        button("Comment", size="1")
                    ),
                    gap="3",
                    mt="3",
                    justify="between",
                ),
                grow="1",
            ),
            gap="3"
        ),
        style={"width": 360},
    )
)
```
